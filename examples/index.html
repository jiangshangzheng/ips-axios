ips-http

查询/订阅指标(单个指标)

ips.http.getData('s0102', {region_id: -212931231}, function(response) {
    console.log(JSON.stringify(response.data))
})

查询/订阅指标(多个指标)

ips.http.getData(['s0102','s0103'], {'s0103':{region_id: -212931231}}, function(ret1,ret2) {
    console.log(JSON.stringify(ret1.data))
    console.log(JSON.stringify(ret1.data))
})
